# Типы данных


```js
// Массив (особый тип объекта)--------------------------------------------------
// Список переменных
var typeArray = ['яблоко', 'вишня', 'апельсин'];
console.log( typeof typeArray ); // object

// Функция (особый тип объекта)-------------------------------------------------
// Набор действий
var typeFunc = function(){ console.log( 'это функция' ); };
console.log( typeof typeFunc ); // function, хотя функция — это тип объекта
```
