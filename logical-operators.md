# Логические операторы

## Или

```js
// Если хотя бы один из аргументов true, то вернёт true, иначе вернёт false
var h = 12, isWeekend = true;
if (h < 10 || h > 18 || isWeekend) { // (false || false || true) = true
  console.log( 'Закрыто' );
}

// Использует «короткий цикл» слева направо: если одно из условий true,
// то следующие значения игнорируются.
var x;
true || (x = 1);
console.log( x ); // undefined, т.к. при проверке уже нашлось true и далее интерпретатор «не смотрел»
false || (x = 1);
console.log( x ); // 1, т.к. при интерпретатор выполнил всю строку (произошло присваивание)

// Возвращает значение, на котором остановился, причем не приведённое к логическому типу
console.log( 1 || 0 );           // 1
console.log( true || 'строка' ); // true
console.log( null || 1 );        // 1
console.log( undefined || 0 );   // 0, т.к. если все значения false, возвращается последнее из них
```

## И

```js
// Если все аргументы true, то вернёт true, иначе вернёт false
var h = 12, isWeekend = true;
if (h === 12 && isWeekend) { // (true && true) = true
  console.log( 'Открыто лишь 1 час!' );
}

// Использует «короткий цикл» слева направо: если одно из условий false,
// то следующие значения игнорируются.
var y;
false && (y = 1);
console.log( y ); // undefined, т.к. при проверке уже нашлось false и далее интерпретатор «не смотрел»
true && (y = 1);
console.log( y ); // 1, т.к. при интерпретатор выполнил всю строку (произошло присваивание)

// Возвращает значение, на котором остановился, причем не приведённое к логическому типу
console.log( 1 && 0 && false);   // 0, т.к. это первый из найденных false и далее интерпретатор «не смотрел»
console.log( true && 'строка' ); // 'строка', т.к. оба значения привелись к true, а это — последнее
console.log( null && 1 );        // null, т.к. он приведён к false и далее интерпретатор «не смотрел»
console.log( undefined && 0 );   // undefined, т.к. он приведён к false и далее интерпретатор «не смотрел»
```

`&&` имеет более высокий приоритет, чем `||`.

```js
console.log( 5 || 1 && 0 ); // 5, т.к. сначала выполнено 1 && 0 (вернулся 0), а потом выполнен 5 || 0 (вернулось 5)
```

## Не

```js
// Получает один аргумент, приводит его к логическому типу, возвращает противоположное значение
console.log( !true ); // false
console.log( !1 );    // false

// Двойное отрицание часто используется для приведения к логическому типу
console.log( !!'строка' ); // true
console.log( !!null );     // false
```
