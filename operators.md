# Операторы (арифметика, присваивание, конкатенация)

У операторов есть [приоритет](https://developer.mozilla.org/ru/docs/Web/JavaScript/Reference/Operators/Operator_Precedence), чем он выше, тем приоритетнее оператор.

К.О.: `=` — тоже оператор (низкоприоритетный).

```js
// Умножение и деление
console.log( 2 * 3); // 6
console.log( 6 / 2); // 3
console.log( '2' * 3);   // 6 (приведение к числу)
console.log( '6' / 2);   // 3 (приведение к числу)
console.log( '6' / '2'); // 3 (приведение к числу)
console.log( 6 / '2цветок'); // NaN

// Бинарный минус
console.log( 2 - 1 ); // 1
console.log( '2' - 1 ); // 1 (приведение к числу)
console.log( 2 - '1' ); // 1 (приведение к числу)
console.log( 6 - '2цветок'); // NaN

// Унарный минус (смена знака)
var x = 1;
console.log( -x ); // -1

// Бинарный плюс
console.log( 1 + 2 ); // 3
console.log( 2 + false ); // 2 (приведение к числу, а false приводится к 0)
console.log( 2 + true );  // 3 (приведение к числу, а true приводится к 1)
console.log( '1' + 2 ); // 12 (приведение к строке, т.к. один из операндов — строка)
console.log( 2 + '1' ); // 21 (приведение к строке, т.к. один из операндов — строка)
console.log( 'hello ' + 'world' ); // hello world
console.log( 'hello ' + false ); // hello false

// Унарный плюс (приведение к числу)
var x = '1', y = '2';
console.log( x + y);   // 12 (конкатенация строк, сработал бинарный плюс)
console.log( +x + +y); // 3 (обе строки сначала приведены к числу, т.к. сначала сработал унарный плюс (приоритет выше), а потом бинарный)

// Остаток от деления
console.log( 5 % 2 ); // 1, т.к. 5 / 2 = 2 целых (2 × 2 = 4 и ещё останется 1)
console.log( 8 % 3 ); // 2, т.к. 8 / 3 = 2 целых (3 × 2 = 6 и ещё останется 2)
console.log( 6 % 3 ); // 0, т.к. 6 / 3 = 2 (без остатка)

// Инкремент и декремент, префиксная форма (приоритетнее мат. операций)
var i = j = 2;
i = ++i; // инкремент (прибавит 1 и вернёт)
j = --j; // декремент (вычтет 1 и вернёт)
console.log(i); // 3
console.log(j); // 1

// Инкремент и декремент, постфиксная форма (приоритетнее мат. операций)
var i = j = 2;
i = i++; // инкремент (сначала вернёт, а потом прибавит 1)
j = j--; // декремент (сначала вернёт, а потом вычтет 1)
console.log(i); // 2
console.log(j); // 2

// Сокращенные мат. операции
var x = 10; console.log( x -= 2); // 8, т.к. эквивалентно x = x - 2
var x = 10; console.log( x += 2); // 12, т.к. эквивалентно x = x + 2
var x = 10; console.log( x *= 2); // 20, т.к. эквивалентно x = x * 2
var x = 10; console.log( x /= 2); // 5, т.к. эквивалентно x = x / 2
var x = 10; console.log( x %= 3); // 1, т.к. эквивалентно x = x % 2
```
